    private void btnSamaDenganActionPerformed(java.awt.event.ActionEvent evt) {                                              
        switch(pilih){
            case 1:
            angka2 = Double.parseDouble(angka);
            total = angka1 + angka2;
            angka = Double.toString(total);
            display.setText(angka);
            break;
            case 2:
            angka2 = Double.parseDouble(angka);
            total = angka1 - angka2;
            angka = Double.toString(total);
            display.setText(angka);
            break;
            case 3:
            angka2 = Double.parseDouble(angka);
            total = angka1 * angka2;
            angka = Double.toString(total);
            display.setText(angka);
            break;
            case 4:
            angka2 = Double.parseDouble(angka);
            total = angka1 / angka2;
            angka = Double.toString(total);
            display.setText(angka);
            break;
            default:
            break;
        }
    }
